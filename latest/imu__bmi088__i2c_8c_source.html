108 void imu_bmi088_event(void)
109 {
110  uint32_t now_ts = get_sys_time_usec();
111 
112  // If the BMI088 I2C transaction has succeeded: convert the data
113  bmi088_i2c_event(&imu_bmi088.bmi);
114 
115  if (imu_bmi088.bmi.gyro_available) {
116  struct Int32Rates rates = {
117  IMU_BMI088_X_SIGN *(int32_t)(imu_bmi088.bmi.data_rates.value[IMU_BMI088_CHAN_X]),
118  IMU_BMI088_Y_SIGN *(int32_t)(imu_bmi088.bmi.data_rates.value[IMU_BMI088_CHAN_Y]),
119  IMU_BMI088_Z_SIGN *(int32_t)(imu_bmi088.bmi.data_rates.value[IMU_BMI088_CHAN_Z])
120  };
121  // unscaled vector
122  RATES_COPY(imu.gyro_unscaled, rates);
123 
124  imu_bmi088.bmi.gyro_available = false;
125 
126  imu_scale_gyro(&imu);
127  AbiSendMsgIMU_GYRO_INT32(IMU_BMI088_ID, now_ts, &imu.gyro);
128  }
129  if (imu_bmi088.bmi.accel_available) {
130  // set channel order
131  struct Int32Vect3 accel = {
132  IMU_BMI088_X_SIGN *(int32_t)(imu_bmi088.bmi.data_accel.value[IMU_BMI088_CHAN_X]),
133  IMU_BMI088_Y_SIGN *(int32_t)(imu_bmi088.bmi.data_accel.value[IMU_BMI088_CHAN_Y]),
134  IMU_BMI088_Z_SIGN *(int32_t)(imu_bmi088.bmi.data_accel.value[IMU_BMI088_CHAN_Z])
135  };
136  // unscaled vector
137  VECT3_COPY(imu.accel_unscaled, accel);
138 
139  imu_bmi088.bmi.accel_available = false;
140 
141  imu_scale_accel(&imu);
142  AbiSendMsgIMU_ACCEL_INT32(IMU_BMI088_ID, now_ts, &imu.accel);
143  }
144 }
