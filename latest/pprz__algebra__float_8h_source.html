107 static inline float float_log_n(float v, float n)
108 {
109  if (fabsf(v) < 1e-4) { // avoid inf
110  return - 1.0E+30;
111  }
112  if (fabsf(n) < 1e-4) { // avoid nan
113  return 0;
114  }
115  return logf(fabsf(v)) / logf(n);
116 }
