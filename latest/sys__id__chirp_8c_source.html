79 static void set_current_chirp_values(void)
80 {
81  if (chirp_active) {
82 #if CHIRP_USE_NOISE
83 
84  float amplitude, noise;
85  for (uint8_t i = 0; i < CHIRP_NB_AXES; i++) {
86  noise = update_first_order_low_pass(&filters[i], rand_gaussian());
87  amplitude = chirp_axis == i ? chirp_noise_stdv_onaxis_ratio * chirp_amplitude : chirp_noise_stdv_offaxis;
88  current_chirp_values[i] = (int32_t)(noise * amplitude);
89  }
90 
91 #endif
92 
93  current_chirp_values[chirp_axis] += (int32_t)(chirp_amplitude * chirp.current_value);
94  } else {
95  for (uint8_t i = 0; i < CHIRP_NB_AXES; i++) {
96  current_chirp_values[i] = 0;
97  }
98  }
99 }

157 void sys_id_chirp_init(void)
158 {
159 #if CHIRP_USE_NOISE
160 
161  init_random();
162 
163 #endif
164 
165  chirp_init(&chirp, chirp_fstart_hz, chirp_fstop_hz, chirp_length_s, get_sys_time_float(), CHIRP_EXPONENTIAL,
166  CHIRP_FADEIN);
167  set_current_chirp_values();
168  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_CHIRP, send_chirp);
169 
170  // Filter cutoff frequency is the chirp maximum frequency
171  float tau = 1 / (chirp_fstop_hz * 2 * M_PI);
172  for (uint8_t i = 0; i < CHIRP_NB_AXES; i++) {
173  init_first_order_low_pass(&filters[i], tau, SYS_ID_CHIRP_RUN_PERIOD, 0);
174  current_chirp_values[i] = 0;
175  }
176 }

191 #endif
